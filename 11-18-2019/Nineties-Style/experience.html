#calculates the result of two numbers and an operand.
input1 = 0
input3 = 0

def takeinputs():
input1 = int(input("Enter the first number for your calculation: "))
input2 = input("Enter the operation desired (+, -, *, /):")
input3 = int(input("Enter the second number for your calculation: "))
return input1, input2, input3

def add(input1, input3):
return (input1 + input3)

def sub(input1, input3):
return (input1 - input3)

def mul(input1, input3):
return (input1 * input3)
def div(input1, input3):
return (input1 / input3)

def calc():
(input1,input2,input3) = takeinputs()
if input2 == "+":
result = add(input1, input3)
elif input2 == "-":
result = sub(input1, input3)
elif input2 == "*":
result = mul(input1, input3)
elif input2 == "/":
result = div(input1, input3)
else:
print("Invalid Operand! Please try again.")
calc()
print(f"{input1} {input2} {input3} equals {result}")
again = input('Press "A" to calculate another equation. press any other key to exit program.').upper()
if(again == 'A'):
calc()
else:
print("Exiting Program!")
exit(1)

#Pool table management application. Saves tables time played and cost in a file called table_logs.txt
#Author: Baylie

import time
import datetime
import os
import json

tables = []
tables_vars = []

class Table:
def __init__(self, table_number, is_occupied):
self.table_number = table_number
self.is_occupied = False
self.cost = 0
self.time_start = 0.0
self.time_end = 0.0
tables.append(self)

def start_time(self):
self.time_start = time.time()
#self.time_started = strftime('%y-%m-%d %H:%M:%S', localtime())
self.time_started = str(datetime.datetime.now())
def end_time(self):
self.time_end = time.time()
#self.time_ended = strftime('%y-%m-%d %H:%M:%S', localtime())
self.time_ended = str(datetime.datetime.now())
self.time_played()

def time_played(self):
self.time_passed = self.time_end - self.time_start
self.minutes_passed = round(self.time_passed / 60,2)
self.cost = self.minutes_passed * .50
#print(end_time(self)-start_time(self))

def start_table(self):
self.start_time()
self.is_occupied = True
print(f'Table {self.table_number} Started at {self.time_started}')

def end_table(self):
self.is_occupied = False
self.end_time()
print(f'Table {self.table_number} Ended at {self.time_ended}')
self.print_to_txt()
self.print_to_json()
def print_to_txt(self):
try:
os.mkdir(f'{datetime.datetime.now():%Y-%m-%d}')
except:
('directory already created!')
with open(f'{datetime.datetime.now():%Y-%m-%d}/table_logs.txt','a') as file_object:
file_object.write(f'Table {self.table_number}: \r START TIME: {self.time_started} \r END TIME: {self.time_ended} \r TOTAL TIME PLAYED: {self.minutes_passed} MINUTES \r TOTAL COST: ${self.cost} \r _______________________________________________ \n')
def print_to_json(self):
try:
os.mkdir(f'{datetime.datetime.now():%Y-%m-%d}')
except:
('directory already created!')
with open(f'{datetime.datetime.now():%Y-%m-%d}/table_logs.json','w') as file_object:
